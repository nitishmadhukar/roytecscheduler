Ext.define('Sch.patches.TouchScroll', {
    extend   : 'Sch.util.Patch',

    requires   : ['Ext.scroll.TouchScroller'],
    target     : 'Ext.scroll.TouchScroller',
    
    minVersion : '5.1.0',

    overrides  : {
        privates : {
            onEvent: function(e) {
                var me = this;

                if (!me[me.listenerMap[e.type]]) return;

                return this.callParent(arguments);
            }
        }
    }
});
